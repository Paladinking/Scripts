

extern fn memmove(uint8* a, uint8* b, uint64 len);
extern fn WriteFile(uint64 h, uint8* b, uint32 to_write, uint32* written, uint8* overlapped) -> bool;


fn fib_rec_u64(uint64 n) -> uint64 {
    if (n < 2) {
        return n;
    }
    return fib_rec_u64(n - 1) + fib_rec_u64(n - 2);
}

fn main() -> uint64 {
    uint8* str = "Hello world\n";

    uint32 w;
    WriteFile(0, str, 12, &w, NULL);

    uint64 ix = 0;
    int32 sum = 0;
    while (str[ix] != 0) {
        sum = sum + str[ix];

        ix = ix + 1;
    }

    return five_sum(sum, 1, 2, 3, 4);
}

fn five_sum(uint32 a, uint16 b, int32 c, int64 d, uint32 e) -> int64 {
    return a + b + c + d + e;
}
