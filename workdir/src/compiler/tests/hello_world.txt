struct Y {
    uint64 n;
    X x;
    X[2] arr;
}

struct X {
    uint64 a;
    int64 b;
    uint16[12] arr;
}

fn main() -> uint64 {
    Y y;
    y.x.a = 123;
    y.x.b = 456;

    X x1 = y.x;

    Y y2;
    y2.x = y.x;


    Y[5] arr;

    uint64 ix = 0;
    while (ix < 5) {
        arr[ix].arr[1].arr[6] = ix;

        ix = ix + 1;
    }

    return arr[2].arr[1].arr[6] + arr[3].arr[1].arr[6] + x1.a + y2.x.b;
}
